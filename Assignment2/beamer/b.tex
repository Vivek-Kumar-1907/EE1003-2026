\begin{frame}{Question}
\textbf{Question}:\\
The iteration scheme
\begin{align}
    x_{n+1} = \sqrt{a}\brak{1+\frac{3a^2}{x_n^2}} - \frac{3a^2}{x_n}, \quad a>0
\end{align}
converges to the real number
\begin{enumerate}
    \item $\sqrt{a}$
    \item $a$
    \item $a\sqrt{a}$
    \item $a^2$
\end{enumerate}
\end{frame}

\begin{frame}[fragile, allowframebreaks]{Solution}
Since this problem is a fixed-point iteration, and we are supposed to find $x = \lim\limits_{n\to\infty} x_n$. So we simply need to find $x$ where
\begin{align}
    x &= g(x) \label{eq:1} \\
    g(x) &= \sqrt{a}\brak{1+\frac{3a^2}{x^2}} - \frac{3a^2}{x}
\end{align}
On simplification and solving, we get
\begin{align}
    x^3 -\sqrt{a}x^2 + 3a^2x - 3a^2\sqrt{a} = 0\\
    x^2(x-\sqrt{a}) + 3a^2(x-\sqrt{a})=0\\
    (x^2+3a^2)(x-\sqrt{a}) =0 \\
    \implies x= \sqrt{a} \label{eq:3}
\end{align}
Hence it converges to $\sqrt{a}$

\begin{center}
$\boxed{\textbf{Option A is the correct answer}}$
\end{center}
\end{frame}

\begin{frame}[fragile, allowframebreaks]{Analyzing conditions for $a$}
    Now, we try to apply Z-Transform in this question:
We cannot directly apply it in this case, due to it being non-linear. We will try to linearize this fixed-point iteration so that we can apply z-transform. Since we know  $\lim\limits_{n\to\infty} x_n = \sqrt{a}$
\begin{align}
x_{n} = \sqrt{a} + \epsilon_n \label{eq:2}\\
x_{n+1} = g(x_{n})\\
\implies \sqrt{a} + \epsilon_{n+1} = g(\sqrt{a} + \epsilon_n)
\end{align}
Since error is small ($\epsilon_n << 1$), we can write
\begin{align}
    g(\sqrt{a} + \epsilon_n) = g(\sqrt{a}) + \epsilon_ng'(\sqrt{a})
\end{align}
Finding $g(\sqrt{a})$ and $g(\sqrt{a})$.
From \eqref{eq:1}, \eqref{eq:3}, $g(\sqrt{a}) = \sqrt{a}$.
\begin{align}
    g'(x) = \frac{-6a^2\sqrt{a}}{x^3} + \frac{3a^2}{x^2}\\
    g'(\sqrt{a}) = -6a + 3a = -3a
\end{align}

\begin{align}
    \epsilon_{n+1} = -3a\epsilon_n
\end{align}

We finally have a linear sequence. It is basically a GP. For the GP to converge and not diverge, intuitively we can say that $|-3a| < 1$.
Anyways we apply Z-transform on both sides. Let E(z) be z-transform of $\epsilon_n$.
\begin{align}
    zE(z) - z\epsilon_n = -3aE(z) \\
    E(z) = \frac{z\epsilon_n}{z+3a}
\end{align}
We find the poles:
\begin{align}
    z + 3a = 0 \implies z = -3a
\end{align}

For the iteration to be stable, and for error sequence to converge to 0, $z$ must lie within the unit circle. 
\begin{align}
    |z| < 1 \implies |a| < \frac{1}{3}
\end{align}
\begin{center}
$\boxed{\mathbf{x_n} \textbf{ converges only if } \mathbf{|a| < 1/3}}$
\end{center}
\end{frame}

import numpy as np
import matplotlib.pyplot as plt
import math

def g_x(a, x):
  return math.sqrt(a)*(1+((3*a**2)/x**2)) - 3*(a**2)/x

n=1001
print('Enter a:')
a=float(input())
x = np.arange(1, n+1)
y = np.zeros(n)
y[0] = 1
for i in range(1,n):
  y[i] = g_x(a, y[i-1])
plt.plot(x, y, label = 'x_n')
#plt.xlim(0, 25)
plt.xlabel('n')
plt.ylabel('x_n')
plt.legend()
plt.title(f'x_n vs n for a = {a}')
print(f'x_n after {n-1} iterations: x_{n-1} = {y[n-1]}')
\end{lstlisting}

import numpy as np
import matplotlib.pyplot as plt
import math

def g_x(a, x):
  return math.sqrt(a)*(1+((3*a**2)/x**2)) - 3*(a**2)/x

n=120
print('Enter a:')
a=float(input())
x = np.arange(1, n+1)
y = np.zeros(n)
y[0] = 1
for i in range(1,n):
  y[i] = g_x(a, y[i-1])
z = np.zeros(n)
for i in range(1, n):
  z[i] = np.log(np.abs(y[i] - y[n-1]))
plt.plot(x, z, label = 'x_n')
plt.xlabel('n')
plt.ylabel(f'log(x_n-{math.sqrt(a)})')
plt.legend()
plt.title(f'log_10(x_n-{math.sqrt(a)}) vs n for a = {a}')
print(f'log_10(x_n-{math.sqrt(a)}) after {n-1} iterations: {z[n-1]}')
\end{lstlisting}

\begin{frame}[fragile, allowframebreaks]{Python Code - Plotting cobweb plot}
    \begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt
import math

def g_x(a, x):
  return math.sqrt(a)*(1+((3*a**2)/x**2)) - 3*(a**2)/x

print('Enter a:')
a = float(input())
x0 = 1.0
n_iterations = 10
true_limit = math.sqrt(a)

x_vals = np.linspace(0.1, 2, 500)
y_vals = g_x(a, x_vals)

plt.plot(x_vals, x_vals, '--', color='gray', label='y = x (Diagonal)')
plt.plot(x_vals, y_vals, color='blue', linewidth=2, label='y = g(x)')

cobweb_x = [x0]
cobweb_y = [0]

current_x = x0

for i in range(n_iterations):
    next_x = g_x(a, current_x)

    cobweb_x.append(current_x)
    cobweb_y.append(next_x)
    
    cobweb_x.append(next_x)
    cobweb_y.append(next_x)
    
    current_x = next_x

plt.plot(cobweb_x, cobweb_y, color='red', alpha=0.7, linewidth=1.5, label='Cobweb Path')

plt.plot(x0, 0, 'go', markersize=8, label=f'Start (x0={x0})')
plt.plot(true_limit, true_limit, 'ko', markersize=8, label=f'True Root (L={true_limit:.3f})')

plt.xlim(0.1, 2)
plt.ylim(0.1, 2)
plt.xlabel('x_n')
plt.ylabel('x_{n+1}')
plt.title(f'Cobweb Plot for a = {a}')
plt.grid(True)
plt.legend()
plt.show()
    \end{lstlisting}
