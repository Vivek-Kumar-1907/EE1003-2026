\textbf{Question}:\\
If F(s) is the Laplace transform of function f(t), then Laplace transform of $\int\limits_0^t f(\tau) d\tau$ is
\begin{enumerate}
    \item $-F(s)$
    \item $-F(s) - f(0)$
    \item $\frac{1}{s}F(s)$
    \item $\int F(s)ds$
\end{enumerate}
\textbf{Solution: }
\begin{align}
\mathscr{L}\{f(t)\} = F(s) = \int_0^{\infty}f(t)e^{-st}dt \\
\end{align}
Let, 
\begin{align}
    g(t) &= \int\limits_0^tf(\tau)d\tau \label{eq:1} \\
    \mathscr{L}\{g'(t)\} &= \int\limits_0^\infty g'(t)e^{-st}dt \\
    &= g(t)e^{-st} \bigg|_{0}^{\infty} - \int\limits_0^\infty g(t)\frac{d}{dt}(e^{-st})dt \\
    &= (0 - g(0)) + s\int\limits_0^\infty g(t)e^{-st}dt \\
    g(0) &= \int\limits_0^0f(\tau)d\tau
         = 0\\
    \implies \mathscr{L}\{g'(t)\} &= s \mathscr{L}\{g(t)\} \\
    \mathscr{L}\{g(t)\} &= \frac{1}{s}\mathscr{L}\{g'(t)\}
\end{align}
Applying fundamental theorem of calculus to \eqref{eq:1},
\begin{align}
    \frac{d}{dt}g(t) = f(t)
\end{align}

So we finally end up with
\begin{align}
    \mathscr{L}\cbrak{\int\limits_0^tf(\tau)d\tau} = \frac{1}{s}\mathscr{L}\{f(t)\} = \frac{F(s)}{s}
\end{align}
