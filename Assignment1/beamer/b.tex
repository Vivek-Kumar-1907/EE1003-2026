\begin{frame}{Question}
If F(s) is the Laplace transform of function f(t), then Laplace transform of $\int\limits_0^t f(\tau) d\tau$ is
\begin{enumerate}
    \item $-F(s)$
    \item $-F(s) - f(0)$
    \item $\frac{1}{s}F(s)$
    \item $\int F(s)ds$
\end{enumerate}
\end{frame}

\begin{frame}{Solution}
\begin{align}
\mathcal{L}\{f(t)\} = F(s) = \int_0^{\infty}f(t)e^{-st}dt
\end{align}
Let, 
\begin{align}
    g(t) &= \int\limits_0^tf(\tau)d\tau \label{eq:1} \\
    \mathcal{L}\{g'(t)\} &= \int\limits_0^\infty g'(t)e^{-st}dt \\
    &= g(t)e^{-st} \bigg|_{0}^{\infty} - \int\limits_0^\infty g(t)\frac{d}{dt}(e^{-st})dt 
\end{align}
\end{frame}

\begin{frame}{Solution}
\begin{align}
    &= (0 - g(0)) + s\int\limits_0^\infty g(t)e^{-st}dt \\
    g(0) &= \int\limits_0^0f(\tau)d\tau
         = 0\\
    \implies \mathcal{L}\{g'(t)\} &= \frac{1}{s} \mathcal{L}\{g(t)\} \\
    \mathcal{L}\{g(t)\} &= \frac{1}{s}\mathcal{L}\{g'(t)\}
\end{align}
\end{frame}

\begin{frame}{Solution}
Applying fundamental theorem of calculus to \eqref{eq:1},
\begin{align}
    \frac{d}{dt}g(t) = f(t)
\end{align}

So we finally end up with
\begin{align}
    \mathcal{L}\cbrak{\int\limits_0^tf(\tau)d\tau} = \frac{1}{s}\mathcal{L}\{f(t)\} = \frac{F(s)}{s}
\end{align}
\begin{center}
$\boxed{\textbf{Option C is the right answer}}$
\end{center}
\end{frame}

\begin{frame}[fragile, allowframebreaks]{Python Code - Computing and plotting laplace transform of given signals}
\begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt
from scipy import integrate
import math
def integrand(t, s, a):
  return (t**a)*np.pow(np.e, -s*t)/math.factorial(a)
n = 100
S = np.linspace(0, 10, num = n)
lt_1 = np.zeros(n)
lt_2 = np.zeros(n)
lt_3 = np.zeros(n)
for i in range(0, n):
  lt_1[i],_ = integrate.quad(integrand, 0, np.inf, args=(S[i], 1))
  lt_2[i],_ = integrate.quad(integrand, 0, np.inf, args=(S[i], 2))
  lt_3[i],_ = integrate.quad(integrand, 0, np.inf, args=(S[i], 3))
plt.plot(S, lt_1, label = 's-transform of f(t) = t -> F(s) = 1/s^2')
plt.plot(S, lt_2, label = 's-transform of f(t) = t^2/2 -> F(s) = 1/s^3')
plt.plot(S, lt_3, label = 's-transform of f(t) = t^3/3 -> F(s) = 1/s^4')
plt.xlim(0, 4)
plt.ylim(0, 4)
plt.legend()
plt.xlabel('s-axis')
plt.ylabel('F(s)')
plt.title('Plot after computing laplace transform')
\end{lstlisting}
\end{frame}
