\begin{frame}{Question}
\textbf{Question}:\\
\begin{align}
    \lim\limits_{x\to\infty} \frac{x-\sin x}{x+\cos x} 
\end{align}
equals
\begin{enumerate}
    \item $1$
    \item $-1$
    \item $\infty$
    \item $0$
\end{enumerate}
\end{frame}

\begin{frame}[fragile, allowframebreaks]{Solution}
The minimum and maximum of the numerator $x-\sin x$ is $x-1$ and $x+1$. The minimum and maximum of the denominator $x+\cos x$ similarly, is $x-1$ and $x+1$. 
\begin{align}
    \frac{x-1}{x+1} &< \frac{x-\sin x}{x+\cos x} < \frac{x+1}{x-1} \\
    \lim\limits_{x\to\infty} \frac{x-1}{x+1} &= \lim\limits_{x\to\infty} \frac{x+1}{x-1} = 1 \text{ (By L'Hospital)}
\end{align}

So by sandwich rule,
\begin{align}
    \lim\limits_{x\to\infty} \frac{x-\sin x}{x+\cos x}  = 1
\end{align}

\begin{center}
    \boxed{\textbf{Option A is the answer: Limit is 1}}
\end{center}
\end{frame}

\begin{frame}[fragile,allowframebreaks]{Python Code - Plotting upper, lower bounds and function }
\begin{lstlisting}[language=Python]
import numpy as np
import matplotlib.pyplot as plt
num = 100
x = np.linspace(0, 75, num = num)

y1 = (x-1)/(x+1)
y3 = (x+1)/(x-1)
y2 = (x-np.sin(x))/(x+np.cos(x))

plt.plot(x, y1, label = 'x-1/x+1')
plt.plot(x, y2, label = 'x-sinx/x+cosx')
plt.plot(x, y3, label = 'x+1/x-1')
plt.ylim(-1.5, 3.5)
plt.legend()
plt.title('Sandwich rule visualisation')
\end{lstlisting}
